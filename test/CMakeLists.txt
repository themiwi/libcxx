function(libcxx_add_tests)
	set(test_srcs "${ARGN}")
	file(RELATIVE_PATH name_prefix
		${CMAKE_SOURCE_DIR} ${CMAKE_CURRENT_SOURCE_DIR})
	string(REGEX REPLACE "[^a-zA-Z0-9;.+-]" "_" exe_prefix "${name_prefix}")

	string(REGEX REPLACE "\\.cpp" "" test_names "${test_srcs}")
	string(REGEX REPLACE "[^a-zA-Z0-9;.+-]" "_" test_exes "${test_names}")

	list(LENGTH test_srcs n)
	math(EXPR n "${n}-1")
	foreach(i RANGE ${n})
		list(GET test_srcs ${i} tsrc)
		list(GET test_names ${i} tname)
		list(GET test_exes ${i} texe)
		set(texe ${exe_prefix}_${texe})
		add_executable(${texe} EXCLUDE_FROM_ALL ${tsrc})
		target_link_libraries(${texe} c++)
		set_target_properties(${texe} PROPERTIES
			COMPILE_FLAGS "${compile_flags}"
			LINK_FLAGS "${link_flags}")
		add_test(NAME ${tname}.build
			COMMAND ${CMAKE_COMMAND} --build ${CMAKE_BINARY_DIR} --target ${texe})
		if(tsrc MATCHES "\\.pass\\.cpp$")
			add_test(NAME ${tname}.run COMMAND ${texe})
			set_tests_properties(${tname}.run PROPERTIES DEPENDS ${tname}.build)
		elseif(tsrc MATCHES "\\.fail\\.cpp$")
			set_tests_properties(${tname}.build PROPERTIES WILL_FAIL TRUE)
		else()
			message(FATAL_ERROR "Invalid test source file name: ${tsrc}")
		endif()
	endforeach()
endfunction()
